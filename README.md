  document.addEventListener('DOMContentLoaded', function() {
    const englishVersion = document.getElementById('english-version');
    const spanishVersion = document.getElementById('spanish-version');
    const englishLink = document.getElementById('english-link');
    const spanishLink = document.getElementById('spanish-link');
    
    // Initialize with English version showing
    englishVersion.style.display = 'block';
    spanishVersion.style.display = 'none';
    
    // Function to show English version
    function showEnglish() {
      englishVersion.style.display = 'block';
      spanishVersion.style.display = 'none';
      localStorage.setItem('language', 'english');
    }
    
    // Function to show Spanish version
    function showSpanish() {
      englishVersion.style.display = 'none';
      spanishVersion.style.display = 'block';
      localStorage.setItem('language', 'spanish');
    }
    
    // Add click event listeners
    englishLink.addEventListener('click', function(e) {
      e.preventDefault();
      showEnglish();
    });
    
    spanishLink.addEventListener('click', function(e) {
      e.preventDefault();
      showSpanish();
    });
    
    // Check if language preference is stored
    const storedLanguage = localStorage.getItem('language');
    if (storedLanguage === 'spanish') {
      showSpanish();
    } else {
      showEnglish();
    }
  });
</script>
